Ort::Env env(ORT_LOGGING_LEVEL_WARNING, "test");
Ort::SessionOptions session_options;
Ort::Session session(env, "model.onnx", session_options);
std::vector input = {1, 2, 3};
std::vector offsets = {0};
Ort::Value input_tensor = Ort::Value::CreateTensor(input.data(), input.size(), {1, 3});
Ort::Value offsets_tensor = Ort::Value::CreateTensor(offsets.data(), offsets.size(), {1});
std::vector inputs = {input_tensor, offsets_tensor};
auto output_tensors = session.Run(Ort::RunOptions{nullptr}, inputs, {"output"});
auto result = output_tensors.front().GetTensorMutableData();
std::cout << result[0] << std::endl;
