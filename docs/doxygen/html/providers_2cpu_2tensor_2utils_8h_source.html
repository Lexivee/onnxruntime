16  static bool Calculate(gsl::span<int64_t> p, const std::vector<int64_t>& dims) {
17  // The pitches is the size of the next inner axis. Aka the amount to move by one of the next inner axis.
18  // For a tensor with shape(2,3,4,5) the values would be: (3*4*5, 4*5, 5, 1)
19  // Note that the outermost '2' is never used, as you never need to move by the entire size of the outermost axis
20 
21  auto tensor_rank = dims.size();
22  auto pitch_rank = p.size();
23  auto padded_rank = pitch_rank - tensor_rank;
24  if (gsl::narrow_cast<ptrdiff_t>(padded_rank) < 0)
25  return false;
26 
27  *(p.rbegin()) = 1; // The innermost axis is 1 (single values)
28  if (tensor_rank > 1) {
29  for (size_t i = tensor_rank - 1; i-- > 0;) {
30  p.operator[](i + padded_rank) = p.operator[](i + 1 + padded_rank) * dims[i + 1];
31  }
32  }
33 
34  if (padded_rank >= 1) {
35  for (size_t i = 0; i < padded_rank; ++i) {
36  if (i == 0)
37  p.operator[](padded_rank - 1) = p.operator[](padded_rank) * dims[0];
38  else
39  p.operator[](padded_rank - 1 - i) = p.operator[](padded_rank - 1);
40  }
41  }
42  return true;
43  }
44 };

46 // This class is to iterate through the axes of an arbitrarily shaped tensor
47 // For example, a tensor with shape (2,3,4) will be iterated in this order:
48 // (0,0,x) (0,1,x) (0,2,x) (1,0,x) (1,1,x) (1,2,x)

53  axis_ = indices_.size();

55  // If a tensor has a shape, but one of the axes is 0 in size, there are no elements, so nothing to iterate
56  if (tensor_.Shape().Size() == 0)
57  running_ = false;
58  }

60  // Returns true if there was a carry to the next axis
61  bool Increment() {
62  if (axis_-- == 0) {
63  running_ = false;
64  return false;
65  }
66 
67  if (++indices_[axis_] != tensor_.Shape()[axis_]) {
68  axis_ = indices_.size();
69  return false;
70  }
71 
72  indices_[axis_] = 0; // Reset the counter for this axis
73  return true; // There was a carry
74  }

77  operator bool() const { return running_; }

86 // A std::vector that holds the number of entries to skip to go to the next axis start given an extent in each axis
87 // This is used by the SliceIterator to iterate over a slice of a tensor
88 struct SliceSkips : std::vector<int64_t> {
89  SliceSkips(const Tensor& tensor, gsl::span<const int64_t> extents)
90  : std::vector<int64_t>(tensor.Shape().NumDimensions(), 0) {
91  auto& dims = tensor.Shape().GetDims();
92  ONNXRUNTIME_ENFORCE(static_cast<ptrdiff_t>(dims.size()) == extents.size());
93  size_t pitch = dims.back();
94  back() = pitch - extents[size() - 1];
95  for (size_t i = size() - 1; i-- > 0;) {
96  auto prevPitch = pitch;
97  pitch *= dims[i];
98  operator[](i) = pitch - prevPitch * extents[i];
99  }
100  }
101 };

103 // This provides easy sequential iteration over a subset of a tensor given a span of starts & etents
104 template <typename T>
105 struct SliceIterator {
106  SliceIterator(const Tensor& tensor, gsl::span<const int64_t> starts, gsl::span<const int64_t> extents)
107  : tensor_(tensor), extents_(extents), skips_(tensor, extents), indices_(extents.size(), 0) {
108  auto& dims = tensor_.Shape().GetDims();
109  ONNXRUNTIME_ENFORCE(static_cast<ptrdiff_t>(dims.size()) == starts.size() && static_cast<ptrdiff_t>(dims.size()) == extents.size());
110 
111  size_t pitch = 1;
112  // Initial skip, so that input_ points to the first element to copy
113  for (size_t i = dims.size(); i-- > 0;) {
114  input_ += pitch * starts[i];
115  pitch *= dims[i];
116  }
117 
118  inner_extent_ = extents_[dims.size() - 1];
119  }
120 
121  void AdvanceOverInnerExtent() {
122  size_t axis = skips_.size() - 1;
123  input_ += skips_[axis];
124  while (axis-- && ++indices_[axis] == extents_[axis]) {
125  indices_[axis] = 0;
126  input_ += skips_[axis];
127  }
128  }
129 
130  const T* operator++(int) {
131  const T* input = input_++;
132  if (++inner_counter_ == inner_extent_) {
133  inner_counter_ = 0;
134  AdvanceOverInnerExtent();
135  }
136  return input;
137  }
138 
139  T* CopyInnermostAxis(T* output) {
140  gsl::copy(gsl::make_span(input_, inner_extent_), gsl::make_span(output, inner_extent_));
141  input_ += inner_extent_;
142  output += inner_extent_;
143  AdvanceOverInnerExtent();
144  return output;
145  }
146 
147  private:
148  const Tensor& tensor_;
149  const T* input_{tensor_.template Data<T>()};
150  gsl::span<const int64_t> extents_;
151  size_t inner_counter_{}, inner_extent_;
152  SliceSkips skips_;
153  std::vector<int64_t> indices_; // There is no index for innermost axis since it's a special case
154 };
