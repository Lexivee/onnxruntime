63  static void NormalizeDims(const TensorShape& x_shape, std::vector<int64_t>& new_dims) {
64  new_dims.clear();

66  if (orig_dims.size() == 4 /*supported size by CUDA*/ ||
67  orig_dims.size() == 5 /*supported size by CUDA*/) {
68  new_dims = orig_dims;

74  auto num_channels = rank > 1 ? orig_dims[1] : 1;
75  auto width = rank > 3 ? orig_dims[3] : 1;
76  auto height = rank > 2 ? orig_dims[2] : 1;
