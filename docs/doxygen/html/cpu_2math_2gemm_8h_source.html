20  Gemm(const OpKernelInfo& info) : OpKernel(info) {
21  int64_t temp;
22  ONNXRUNTIME_ENFORCE(info.GetAttr<int64_t>("transA", &temp).IsOK());
23  trans_A_ = temp == 0 ? CblasNoTrans : CblasTrans;
24 
25  ONNXRUNTIME_ENFORCE(info.GetAttr<int64_t>("transB", &temp).IsOK());
26  trans_B_ = temp == 0 ? CblasNoTrans : CblasTrans;
27 
28  ONNXRUNTIME_ENFORCE(info.GetAttr<float>("alpha", &alpha_).IsOK());
29  ONNXRUNTIME_ENFORCE(info.GetAttr<float>("beta", &beta_).IsOK());
30  }

32  Status Compute(OpKernelContext* context) const override {
33  const auto X = context->Input<Tensor>(0);
34  const auto W = context->Input<Tensor>(1);
35  const auto B = context->Input<Tensor>(2);
36  GemmHelper helper(X->Shape(), trans_A_ != CblasNoTrans, W->Shape(), trans_B_ != CblasNoTrans, B->Shape());
37 
38  if (!helper.State().IsOK())
39  return helper.State();
40 
41  int64_t M = helper.M();
42  int64_t N = helper.N();
43  int64_t K = helper.K();
44  auto Y = context->Output(0, TensorShape({M, N}));
45 
46  //bias
47  // Todo: we might should move this part into math::gemm to let eigen
48  // have better chance to further optimize it.
49  if (beta_ != 0) {
50  auto output_mat = EigenMatrixMapRowMajor<T_Y>(
51  Y->template MutableData<T_Y>(),
52  M,
53  N);
54  output_mat.setZero();
55 
56  auto& b_shape = B->Shape();
57  // if B is (), (1,) or (1, 1), add the scalar
58  if (b_shape.Size() == 1) {
59  output_mat.array() += *(B->template Data<T_B>());
60  }
61  // B is (N,)
62  else if (b_shape.NumDimensions() == 1) {
63  auto bias_vec = ConstEigenVectorMap<T_B>(
64  B->template Data<T_B>(),
65  N);
66  output_mat.rowwise() += bias_vec.transpose();
67  } else if (b_shape.NumDimensions() == 2) {
68  // B is (M, 1)
69  if (b_shape[1] == 1) {
70  auto bias_vec = ConstEigenVectorMap<T_B>(
71  B->template Data<T_B>(),
72  M);
73  output_mat.colwise() += bias_vec;
74  }
75  // B is (1, N)
76  else if (b_shape[0] == 1) {
77  auto bias_vec = ConstEigenVectorMap<T_B>(
78  B->template Data<T_B>(),
79  N);
80  output_mat.rowwise() += bias_vec.transpose();
81  }
82  // B is (M, N), no broadcast needed.
83  else {
84  auto bias_mat = ConstEigenMatrixMapRowMajor<T_B>(
85  B->template Data<T_B>(),
86  M,
87  N);
88  output_mat += bias_mat;
89  }
90  }
91  }
92 
93  // W * x
94  math::Gemm<T_X, CPUMathUtil>(
95  trans_A_,
96  trans_B_,
97  M,
98  N,
99  K,
100  alpha_,
101  X->template Data<T_X>(),
102  W->template Data<T_W>(),
103  beta_,
104  Y->template MutableData<T_Y>(),
105  &CPUMathUtil::Instance());
106 
107  return Status::OK();
108  }
