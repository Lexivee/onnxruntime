44 // It provides API to register rewrite rules, and API to apply for

47 // Represents a IGraphTransformer determined by a set of rewrite-rules.
48 // The transformer will apply all the rewrite-rules iteratively as
49 // determined by the underlying rewriting-strategy.
50 // Several rewriting-strategies are possible when traversing the graph and applying
51 // rewrite rules, each with different tradeoffs. At the moment, we define one
52 // that performs top-down traversal of nodes.
53 // TODO: Is a bottom-up traversal more efficient?
54 // TODO: Is it worth adding the max number of passes a rule should be applied for?
55 // TODO: We need to define a contract about whether a rewrite rule is allowed to leave
56 // the graph in an inconsistent state (this will determine when and where we will be
57 // calling resolve().
58 class RuleBasedGraphTransformer : public GraphTransformer {
59  public:
60  RuleBasedGraphTransformer(const std::string& name, const std::string& desc) : GraphTransformer(name, desc) {}
61 
62  // Register a rewriting rule.
63  // TODO (revisit needed): Using OpSignature* here will ask that OpSignature
64  // should be stored globally. Otherwise, there will be multiple addresses/pointers
65  // for the same operator or function. To avoid this, we may use OpSignature ID
66  // as the key, which should be name_domain_version.
67  // We will use the string type instead of the OpSchema for now. We should probably
68  // add a version as well.
69  Status Register(const std::string& op_type, std::unique_ptr<RewriteRule> rule);
70 
71  // Returns true if there are rules registered for this op_type.
72  bool HasRules(const std::string& op_type) const {
73  return op_to_rules_.count(op_type) > 0;
74  }
75 
76  // Returns a reference to the vector that contains all rewrite rules registered
77  // for this operator. It assumes that there are registered rules, therefore HasRules
78  // should be called before.
79  const std::vector<std::unique_ptr<RewriteRule>>& GetRewriteRules(const std::string& op_type) const {
80  return op_to_rules_.at(op_type);
81  }
82 
83  private:
84  using RewriteRuleSet = std::unordered_map<std::string, std::vector<std::unique_ptr<RewriteRule>>>;
85 
86  RewriteRuleSet op_to_rules_;
87 };
