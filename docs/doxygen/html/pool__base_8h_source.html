118  strides_.resize(kernel_shape_.size(), 1);

160  inline void InferOutputSize(const std::vector<int64_t>& input_dims,
161  std::vector<int64_t>* output_dims,
162  std::vector<int64_t>* pads) const {
163  ONNXRUNTIME_ENFORCE(input_dims.size() >= 2);
164  if (global_pooling_) {
165  output_dims->assign(input_dims.size() - 2, 1);
166  } else {
167  for (size_t dim = 0; dim < input_dims.size() - 2; ++dim) {
168  int64_t dim_size = 0;
169  ComputeSizeAndPad(static_cast<int>(input_dims[dim + 2]),
170  strides_[dim],
171  kernel_shape_[dim],
172  &pads->at(dim),
173  &pads->at(input_dims.size() + dim - 2),
174  &dim_size);
175  output_dims->push_back(dim_size);
176  }
177  }

180  inline void ComputeSizeAndPad(const int64_t in_size,
181  const int64_t stride,
182  const int64_t kernel,
183  int64_t* pad_head,
184  int64_t* pad_tail,
185  int64_t* out_size) const {
186  if (auto_pad_ != AutoPadType::NOTSET) {
187  switch (auto_pad_) {
188  case AutoPadType::VALID:
189  *pad_head = 0;
190  *pad_tail = 0;
191  *out_size = (in_size - kernel) / stride + 1;
192  break;
193  case AutoPadType::SAME_LOWER: {
194  int64_t legacy_target_size = (in_size + stride - 1) / stride;
195  int64_t pad_needed = (legacy_target_size - 1) * stride + kernel - in_size;
196  *pad_head = (pad_needed + 1) / 2;
197  *pad_tail = pad_needed - *pad_head;
198  *out_size = (in_size + pad_needed - kernel) / stride + 1;
199  break;
200  }
201  case AutoPadType::SAME_UPPER: {
202  int64_t legacy_target_size = (in_size + stride - 1) / stride;
203  int64_t pad_needed = (legacy_target_size - 1) * stride + kernel - in_size;
204  *pad_head = pad_needed / 2;
205  *pad_tail = pad_needed - *pad_head;
206  *out_size = (in_size + pad_needed - kernel) / stride + 1;
207  break;
208  }
209  default: {
210  ONNXRUNTIME_THROW("Unsupported AutoPad Type.");
211  }
212  }
213  } else {
214  *out_size = static_cast<int64_t>(
215  static_cast<float>(in_size + *pad_head + *pad_tail - kernel) / stride + 1);
216  }
217  }
