12 // The interface for arena which manage memory allocations
13 // Arena will hold a pool of pre-allocate memories and manage their lifecycle.
14 // Need an underline IResourceAllocator to allocate memories.
15 // The setting like max_chunk_size is init by IDeviceDescriptor from resource allocator
16 class IArenaAllocator : public IAllocator {
17  public:
18  ~IArenaAllocator() override = default;
19  // Alloc call need to be thread safe.
20  void* Alloc(size_t size) override = 0;
21  // The chunck allocated by Reserve call won't be reused with other request.
22  // It will be return to the devices when it is freed.
23  // Reserve call need to be thread safe.
24  virtual void* Reserve(size_t size) = 0;
25  // Free call need to be thread safe.
26  void Free(void* p) override = 0;
27  virtual size_t Used() const = 0;
28  virtual size_t Max() const = 0;
29  const ONNXRuntimeAllocatorInfo& Info() const override = 0;
30  // allocate host pinned memory?
31 };
