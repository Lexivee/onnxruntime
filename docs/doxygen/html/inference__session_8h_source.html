36 struct SessionOptions {

39 
40  // enable profiling for this session.
41  bool enable_profiling = false;
42 
43  // enable the memory pattern optimization.
44  // The idea is if the input shapes are the same, we could trace the internal memory allocation
45  // and generate a memory pattern for future request. So next time we could just do one allocation
46  // with a big chunk for all the internal memory allocation.
47  bool enable_mem_pattern = true;
48 
49  // enable the memory arena on CPU
