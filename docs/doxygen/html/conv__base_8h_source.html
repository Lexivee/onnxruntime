15 // helper function
16 template <bool ForceSymmetricAutoPadding>
17 Status ComputePadAndOutputShape(
18  const int64_t in_dim,
19  const int64_t stride,
20  const int64_t kernel,
21  const int64_t dilation,
22  AutoPadType pad_type,
23  int64_t* pad_head,
24  int64_t* pad_tail,
25  int64_t* out_dim) {
26  const int64_t dkernel = dilation * (kernel - 1) + 1;
27 
28  if (pad_type == AutoPadType::NOTSET) {
29  *out_dim = static_cast<int64_t>(static_cast<float>(in_dim + *pad_head + *pad_tail - dkernel) / stride + 1);
30  } else {
31  switch (pad_type) {
32  case AutoPadType::VALID:
33  *pad_head = 0;
34  *pad_tail = 0;
35  *out_dim = (in_dim - dkernel) / stride + 1;
36  break;
37  case AutoPadType::SAME_UPPER:
38  case AutoPadType::SAME_LOWER: {
39  ONNXRUNTIME_ENFORCE(dilation == 1, "Dilation not supported for AutoPadType::SAME_UPPER or AutoPadType::SAME_LOWER.");
40  int64_t legacy_target_size = (in_dim + stride - 1) / stride;
41  int64_t pad_needed = (legacy_target_size - 1) * stride + kernel - in_dim;
42  *out_dim = (in_dim + pad_needed - dkernel) / stride + 1;
43 
44  // make sure padding is symmetric
45  if (ForceSymmetricAutoPadding)
46  pad_needed = math::roundUpPow2<int64_t, 2>(pad_needed);
47 
48  if (pad_type == AutoPadType::SAME_LOWER) {
49  *pad_head = (pad_needed + 1) / 2;
50  } else {
51  *pad_head = pad_needed / 2;
52  }
53  *pad_tail = pad_needed - *pad_head;
54  } break;
55  default:
56  return Status(common::ONNXRUNTIME, common::INVALID_ARGUMENT, "pad type not supported.");
57  }
58  }
59  return Status::OK();
60 }

119  Status ValidateInputShape(const Tensor* X, const Tensor* W) const {
120  const int64_t C = X->Shape()[1];
121  const int64_t M = W->Shape()[0];
122 
123  if (X->Shape().NumDimensions() != W->Shape().NumDimensions()) {
124  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, FAIL, "X num_dims does not match W num_dims.",
125  " X: ", X->Shape().ToString().c_str(),
126  " W: ", W->Shape().ToString().c_str());
127  }
128 
129  if (C != W->Shape()[1] * group_) {
130  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, FAIL, "Input channels C is not equal to kernel channels * group.",
131  " C: ", C,
132  " kernel channels: ", W->Shape()[1],
133  " group: ", group_);
134  }
135 
136  if (M % group_ != 0) {
137  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, FAIL, "Output channels M is not divisible by group.",
138  " M: ", M,
139  " group: ", group_);
140  }
141  return Status::OK();

144  template <bool ForceSymmetricAutoPadding = false>
145  Status InferOutputShape(const TensorShape& input_shape,
146  const std::vector<int64_t>& kernel_shape,
147  const std::vector<int64_t>& strides,
148  const std::vector<int64_t>& dilations,
149  std::vector<int64_t>* pads,
150  std::vector<int64_t>* output_shape) const {
151  int rank = gsl::narrow_cast<int>(input_shape.NumDimensions());
152  for (int dim = 0; dim < rank; ++dim) {
153  if (dim >= strides.size() || dim >= kernel_shape.size() ||
154  dim >= dilations.size() || dim >= pads->size() ||
155  rank + dim >= pads->size()) {
156  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, FAIL, "Out of bound access to array");
157  }
158  int64_t dim_size = 0;
159  ONNXRUNTIME_RETURN_IF_ERROR(ComputePadAndOutputShape<ForceSymmetricAutoPadding>(
160  input_shape[dim],
161  strides[dim],
162  kernel_shape[dim],
163  dilations[dim],
164  auto_pad_,
165  &pads->at(dim),
166  &pads->at(input_shape.NumDimensions() + dim),
167  &dim_size));
168  if (dim_size <= 0) {
169  return Status(common::ONNXRUNTIME, common::INVALID_ARGUMENT, "Invalid input shape: " + input_shape.ToString());
170  }
171  output_shape->push_back(dim_size);
172  }
173  return Status::OK();
174  }
