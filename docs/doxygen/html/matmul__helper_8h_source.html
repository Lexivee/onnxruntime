10 class MatMulComputeHelper {
11  public:
12  Status Compute(const TensorShape& left_shape, const TensorShape& right_shape) {
13  // Following numpy.matmul for shape inference:
14  // https://docs.scipy.org/doc/numpy/reference/generated/numpy.matmul.html
15  // The behavior depends on the arguments in the following way.
16  // * If both arguments are 2 - D they are multiplied like conventional matrices.
17  // * If either argument is N - D, N > 2, it is treated as a stack of matrices residing in the last two indexes and broadcast accordingly.
18  // * If the first argument is 1 - D, it is promoted to a matrix by prepending a 1 to its dimensions.After matrix multiplication the prepended 1 is removed.
19  // * If the second argument is 1 - D, it is promoted to a matrix by appending a 1 to its dimensions.After matrix multiplication the appended 1 is removed.
20 
21  size_t left_num_dims = left_shape.NumDimensions();
22  size_t right_num_dims = right_shape.NumDimensions();
23  ONNXRUNTIME_RETURN_IF_NOT(left_num_dims >= 1 && right_num_dims >= 1);
24 
25  // special case for right_shape being 2D and left_shape > 2D by flattening left_shape to 2D
26  // note that padding 1s in front of the right shape can be flattened too
27  if (left_num_dims >= 2 && right_num_dims >= 2 &&
28  right_shape.SizeToDimension(right_num_dims - 1) == right_shape[right_num_dims - 2]) {
29  M_ = left_shape.SizeToDimension(left_num_dims - 1);
30  K_ = left_shape[left_num_dims - 1];
31  N_ = right_shape[right_num_dims - 1];
32  std::vector<int64_t> output_dims = left_shape.GetDims();
33  output_dims[left_num_dims - 1] = N_;
34  output_shape_ = TensorShape(output_dims);
35  output_offsets_ = {0};
36  left_offsets_ = {0};
37  right_offsets_ = {0};
38  ONNXRUNTIME_RETURN_IF_NOT(K_ == right_shape[right_num_dims - 2], "MatMul dimension mismatch");
39  return Status::OK();
40  }
41 
42  bool has_1D_input = (left_num_dims == 1 || right_num_dims == 1);
43 
44  size_t num_input_dims = std::max(left_num_dims, right_num_dims);
45 
46  // use padded dims to compute matrix offsets, right 1D would be padded
47  size_t num_dims_with_pad = num_input_dims + (right_num_dims == 1 ? 1 : 0);
48 
49  // output shape would squeeze the reduced 1D dimension
50  size_t num_output_dims = num_input_dims - (has_1D_input ? 1 : 0);
51 
52  left_padded_dims_ = std::vector<int64_t>(num_dims_with_pad, 1);
53  right_padded_dims_ = std::vector<int64_t>(num_dims_with_pad, 1);
54 
55  if (right_num_dims == 1) {
56  // right padded to (1,...,K,1)
57  right_padded_dims_[num_dims_with_pad - 2] = right_shape[0];
58 
59  if (num_input_dims >= 2) {
60  // left padded to (...,1,K)
61  left_shape.CopyDims(&left_padded_dims_[0], left_num_dims - 1);
62  left_padded_dims_[num_dims_with_pad - 1] = left_shape[left_num_dims - 1];
63  } else {
64  // pad 1 in the front
65  left_shape.CopyDims(&left_padded_dims_[num_dims_with_pad - left_num_dims], left_num_dims);
66  }
67  } else {
68  // pad 1 in the front for left
69  left_shape.CopyDims(&left_padded_dims_[num_dims_with_pad - left_num_dims], left_num_dims);
70  // pad 1 in the front for right
71  right_shape.CopyDims(&right_padded_dims_[num_dims_with_pad - right_num_dims], right_num_dims);
72  }
73 
74  // validate input shape and generate output shape
75  std::vector<int64_t> output_dims(num_output_dims);
76 
77  // broadcasting for all output dims except last two
78  for (int idx_dim = 0; idx_dim < num_dims_with_pad - 2; ++idx_dim) {
79  output_dims[idx_dim] = std::max(left_padded_dims_[idx_dim], right_padded_dims_[idx_dim]);
80  if (left_padded_dims_[idx_dim] != output_dims[idx_dim])
81  ONNXRUNTIME_RETURN_IF_NOT(left_padded_dims_[idx_dim] == 1, "left operand cannot broadcast on dim ", idx_dim);
82  if (right_padded_dims_[idx_dim] != output_dims[idx_dim])
83  ONNXRUNTIME_RETURN_IF_NOT(right_padded_dims_[idx_dim] == 1, "right operand cannot broadcast on dim ", idx_dim);
84  }
85 
86  M_ = has_1D_input ? 1 : left_shape[left_num_dims - 2];
87  K_ = left_shape[left_num_dims - 1];
88  N_ = (right_num_dims == 1) ? 1 : right_shape[right_num_dims - 1];
89 
90  if (!has_1D_input) {
91  ONNXRUNTIME_RETURN_IF_NOT(K_ == right_shape[right_num_dims - 2], "MatMul dimension mismatch");
92  // left (...M x K), right (...K x N), output (...M x N)
93  ONNXRUNTIME_RETURN_IF_NOT(num_dims_with_pad == num_output_dims);
94  output_dims[num_output_dims - 2] = M_;
95  output_dims[num_output_dims - 1] = N_;
96  } else {
97  if (num_output_dims == 0) {
98  // for left and right being both vector, output is scalar thus no shape
99  ONNXRUNTIME_RETURN_IF_NOT(M_ == 1 && N_ == 1);
100  } else {
101  if (left_num_dims == 1) {
102  ONNXRUNTIME_RETURN_IF_NOT(num_dims_with_pad - 1 == num_output_dims);
103  ONNXRUNTIME_RETURN_IF_NOT(K_ == right_shape[right_num_dims - 2], "MatMul dimension mismatch");
104  // left (K), right (...K,N), output (...N)
105  output_dims[num_output_dims - 1] = N_;
106  } else {
107  ONNXRUNTIME_RETURN_IF_NOT(num_dims_with_pad - 2 == num_output_dims);
108  ONNXRUNTIME_RETURN_IF_NOT(K_ == right_shape[0], "MatMul dimension mismatch");
109  // left(...K), right (K), output (...), already assigned
110  }
111  }
112  }
113 
114  // assign shape
115  output_shape_ = TensorShape(output_dims);
116 
117  // compute broadcast offsets
118  ComputeBroadcastOffsets();
119 
120  return Status::OK();
121  }

123  private:
124  void ComputeBroadcastOffsets() {
125  num_broadcasted_dims_ = left_padded_dims_.size() - 2;
126 
127  if (num_broadcasted_dims_ == 0) {
128  left_offsets_ = {0};
129  right_offsets_ = {0};
130  output_offsets_ = {0};
131  return;
132  }
133 
134  left_mat_size_ = M_ * K_;
135  right_mat_size_ = K_ * N_;
136  output_mat_size_ = M_ * N_;
137 
138  // stride in mats and dims for broadcasting
139  left_padded_strides_.resize(num_broadcasted_dims_);
140  right_padded_strides_.resize(num_broadcasted_dims_);
141  output_broadcast_strides_.resize(num_broadcasted_dims_);
142  output_broadcast_dims_.resize(num_broadcasted_dims_);
143  for (size_t i = num_broadcasted_dims_; i > 0; --i) {
144  size_t idx = i - 1;
145  output_broadcast_dims_[idx] = std::max(left_padded_dims_[idx], right_padded_dims_[idx]);
146  output_broadcast_strides_[idx] = ((i == num_broadcasted_dims_) ? 1 : output_broadcast_strides_[idx + 1] * output_broadcast_dims_[idx + 1]);
147  left_padded_strides_[idx] = ((i == num_broadcasted_dims_) ? 1 : left_padded_strides_[idx + 1] * left_padded_dims_[idx + 1]);
148  right_padded_strides_[idx] = ((i == num_broadcasted_dims_) ? 1 : right_padded_strides_[idx + 1] * right_padded_dims_[idx + 1]);
149  }
150 
151  size_t num_offsets = output_broadcast_dims_[0] * output_broadcast_strides_[0];
152  left_offsets_.resize(num_offsets);
153  right_offsets_.resize(num_offsets);
154  output_offsets_.resize(num_offsets);
155 
156  RecursiveFill(0, 0, 0, 0);
157  }

159  void
160  RecursiveFill(size_t idx_dim, size_t idx_left, size_t idx_right, size_t idx_out) {
161  if (idx_dim == num_broadcasted_dims_) {
162  left_offsets_[idx_out] = idx_left * left_mat_size_;
163  right_offsets_[idx_out] = idx_right * right_mat_size_;
164  output_offsets_[idx_out] = idx_out * output_mat_size_;
165  } else {
166  auto left_dim = left_padded_dims_[idx_dim];
167  auto right_dim = right_padded_dims_[idx_dim];
168  auto output_dim = output_broadcast_dims_[idx_dim];
169  for (int64_t i = 0; i < output_dim; ++i) {
170  RecursiveFill(idx_dim + 1,
171  idx_left + i * (left_dim == 1 ? 0 : left_padded_strides_[idx_dim]),
172  idx_right + i * (right_dim == 1 ? 0 : right_padded_strides_[idx_dim]),
173  idx_out + i * output_broadcast_strides_[idx_dim]);
174  }
175  }
176  }
