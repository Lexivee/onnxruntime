20  Status ValidateInput(const Tensor* X) const {
21  if (border_.size() < 4) {
22  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT,
23  "Attribute border needs to be specified with four border elements, got ", border_.size());
24  }
25 
26  const auto dims = X->Shape().GetDims();
27 
28  if (dims.size() < 4) {
29  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT,
30  "Input is expected to have four dimensions corresponding to [N,C,H,W], got ", dims.size());
31  }
32 
33  const int64_t H = dims[2];
34  const int64_t W = dims[3];
35 
36  // find the cropped region, and copy it to the destination matrix
37  int64_t leftBorder = border_[0],
38  topBorder = border_[1],
39  rightBorder = border_[2],
40  bottomBorder = border_[3];
41 
42  if (H < topBorder + bottomBorder) {
43  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT, "Input's height (", H, ") needs to be greater than the topBorder (", topBorder, ") + bottomBorder (", bottomBorder, ")");
44  }
45 
46  if (W < leftBorder + rightBorder) {
47  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT, "Input's width (", W, ") needs to be greater than the leftBorder (", leftBorder, ") + rightBorder (", rightBorder, ")");
48  }
49 
50  int64_t bottomLimit = H - bottomBorder;
51  int64_t rightLimit = W - rightBorder;
52 
53  // scale = (height, width)
54  if (!scale_.empty()) {
55  bottomLimit = topBorder + scale_[0];
56  rightLimit = leftBorder + scale_[1];
57 
58  if (H < bottomLimit) {
59  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT,
60  "Input's height (", H, ") needs to be greater than the topBorder (", topBorder, ") + scale_[0] (", scale_[0], ")");
61  }
62 
63  if (W < rightLimit) {
64  return ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT, "Input's width (", W, ") needs to be greater than the leftBorder (", leftBorder, ") + scale_[1] (", scale_[1], ")");
65  }
66  }
67 
68  return Status::OK();
69  }
70 
71  const std::vector<int64_t> border_; // (leftBorder, topBorder, rightBorder, bottomBorder)
72  const std::vector<int64_t> scale_; // (height, width)
