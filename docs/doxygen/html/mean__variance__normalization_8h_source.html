20  Status Compute(OpKernelContext* context) const override {
21  const Tensor* X = context->Input<Tensor>(0);
22  if (X == nullptr) return Status(common::ONNXRUNTIME, common::FAIL, "input count mismatch");
23 
24  const auto dims = X->Shape().GetDims();
25 
26  if (dims.size() < 4) {
27  return Status(common::ONNXRUNTIME, common::INVALID_ARGUMENT,
28  "Input is expected to have four dimensions corresponding to [N,C,H,W]");
29  }
30 
31  const int64_t N = dims[0];
32  const int64_t C = dims[1];
33  const int64_t H = dims[2];
34  const int64_t W = dims[3];
35 
36  Tensor* Y = context->Output(0, TensorShape({N, C, H, W}));
37  const T* Xdata = X->template Data<T>();
38  T* Ydata = Y->template MutableData<T>();
39 
40  const int64_t sample_size = H * W;
41  Eigen::Array<float, Eigen::Dynamic, 1> mean(C, 1);
42  Eigen::Array<float, Eigen::Dynamic, 1> var(C, 1);
43  mean.setZero();
44  var.setZero();
45 
46  ConstEigenArrayMap<T> X_arr(Xdata, sample_size, N * C);
47  for (int nc = 0; nc < N * C; ++nc) {
48  mean(nc % C) += X_arr.col(nc).sum();
49  }
50  mean /= gsl::narrow_cast<T>(N * sample_size);
51  for (int64_t nc = 0; nc < N * C; ++nc) {
52  var(nc % C) += (X_arr.col(nc) - mean(nc % C)).matrix().squaredNorm();
53  }
54  var /= gsl::narrow_cast<T>(N * sample_size);
55 
56  Eigen::Array<T, Eigen::Dynamic, 1> inv_std;
57  EigenArrayMap<T> Y_arr(Ydata, sample_size, N * C);
58 
59  if (across_channels_) {
60  // m_c = sum(m_i) / n
61  float global_mean = mean.mean();
62 
63  // var_c = [(var_1 + (m_1 - m_c)^2) + ... + (var_n + (m_n - m_c)^2)] / n
64  // = [sum(var_i) + squared_norm(m_i - m_c)] / n
65  float global_var = ((mean - global_mean).matrix().squaredNorm() + var.sum()) / C;
66 
67  // For across channels we can directly use eigen because global_mean and global_var
68  // are just scalars.
69  if (!normalize_variance_) {
70  Y_arr = X_arr - global_mean;
71  } else {
72  float inv_std_scalar = 1 / std::sqrt(global_var);
73  Y_arr = (X_arr - global_mean) * inv_std_scalar;
74  }
75  } else {
76  if (!normalize_variance_) {
77  // inv_std = 1
78  for (int64_t nc = 0; nc < N * C; ++nc) {
79  // y = (x - mean)
80  Y_arr.col(nc) = (X_arr.col(nc) - mean(nc % C));
81  }
82  } else {
83  inv_std = var.sqrt().inverse();
84  for (int64_t nc = 0; nc < N * C; ++nc) {
85  // y = (x - mean) * (inv_std)
86  Y_arr.col(nc) = (X_arr.col(nc) - mean(nc % C)) * inv_std(nc % C);
87  }
88  }
89  }
90  return Status::OK();
91  }
92 
93  protected:
94  int64_t across_channels_;
95  int64_t normalize_variance_;
96 };
