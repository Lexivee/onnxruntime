127  template <typename T>
128  static IAllocatorUniquePtr<T> MakeUniquePtr(std::shared_ptr<IAllocator> allocator, size_t count_or_bytes) {
129  if (allocator == nullptr) return nullptr;
130  // for now limit to fundamental types. we could support others, but to do so either we or the caller
131  // needs to call the dtor for the objects, for buffers allocated on device we don't have destructor
132  //static_assert(std::is_fundamental<T>::value, "Fundamental type required as no destructors are called.");
133 
134  size_t alloc_size = count_or_bytes;
135 
136  // if T is not void, 'count_or_bytes' == number of items so allow for that
137  if (!std::is_void<T>::value) {
138  // sizeof(void) isn't valid, but the compiler isn't smart enough to ignore that this line isn't
139  // reachable if T is void. use std::conditional to 'use' void* in the sizeof call
140  if (!CalcMemSizeForArray(count_or_bytes, sizeof(typename std::conditional<std::is_void<T>::value, void*, T>::type),
141  &alloc_size)) return nullptr;
142  }
143  return IAllocatorUniquePtr<T>{
144  static_cast<T*>(allocator->Alloc(alloc_size)), // allocate
145  [=](T* ptr) { allocator->Free(ptr); }}; // capture IAllocator so it's always valid, and use as deleter
146  }

149 template <size_t alignment>
150 bool IAllocator::CalcMemSizeForArrayWithAlignment(size_t nmemb, size_t size, size_t* out) noexcept {
151  static constexpr size_t max_allowed = (static_cast<size_t>(1) << (static_cast<size_t>(std::numeric_limits<size_t>::digits >> 1))) - alignment;
152  static constexpr size_t max_size = std::numeric_limits<size_t>::max() - alignment;
153  static constexpr size_t alignment_mask = alignment - 1;
154  //Indeed, we only need to check if max_size / nmemb < size
155  //max_allowed is for avoiding unnecessary DIV.
156  if (nmemb >= max_allowed && max_size / nmemb < size) {
157  return false;
158  } else if (size >= max_allowed &&
159  nmemb > 0 && max_size / nmemb < size) {
160  return false;
161  }
162  if (alignment == 0)
163  *out = size * nmemb;
164  else
165  *out = (size * nmemb + alignment_mask) & ~static_cast<size_t>(alignment_mask);
166  return true;
167 }
