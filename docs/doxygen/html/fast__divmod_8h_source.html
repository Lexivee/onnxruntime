66  void find_magic_numbers() {
67  // special case for d = 1, -1
68  if (d_ == 1) {
69  M_ = 0;
70  s_ = 0;
71  a_ = 1;
72  return;
73  } else if (d_ == -1) {
74  M_ = 0;
75  s_ = -1;
76  a_ = -1;
77  return;
78  }
79  // general case
80  const unsigned two31 = 0x80000000;
81  unsigned abs_d = (d_ == 0) ? 1 : abs(d_);
82  unsigned t = two31 + ((unsigned)d_ >> 31); // t = 2^31 + (d < 0) ? 1 : 0
83  unsigned abs_nc = t - 1 - (t % abs_d); // |n_c| = t - 1 - rem(t, |d|)
84  int p = 31;
85  unsigned q1 = two31 / abs_nc; // Init q_1 = 2^31 / |n_c|
86  unsigned r1 = two31 - q1 * abs_nc; // Init r_1 = rem(q_1, |n_c|)
87  unsigned q2 = two31 / abs_d; // Init q_2 = 2^31 / |d|
88  unsigned r2 = two31 - q2 * abs_d; // Init r_2 = rem(q_2, |d|)
89 
90  unsigned delta;
91  // iterate p until
92  // 2^p < n_c * (d - rem(2^p, d)) is satisfied
93  do {
94  ++p;
95  q1 *= 2;
96  r1 *= 2;
97 
98  if (r1 >= abs_nc) {
99  q1 += 1;
100  r1 -= abs_nc;
101  }
102  q2 *= 2;
103  r2 *= 2;
104 
105  if (r2 >= abs_d) {
106  q2 += 1;
107  r2 -= abs_d;
108  }
109  delta = abs_d - r2;
110  } while (q1 < delta ||
111  (q1 == delta && r1 == 0));
112 
113  // store magic numbers
114  M_ = q2 + 1;
115  if (d_ < 0) M_ = -M_;
116  s_ = p - 32;
117 
118  // generate sentinel for correct adds / subs
119  // "generate the add if d > 0 and M < 0"
120  if ((d_ > 0) && (M_ < 0)) a_ = 1;
121  // "generate the sub if d < 0 and M > 0"
122  else if ((d_ < 0) && (M_ > 0))
123  a_ = -1;
124  // Otherwise no add / sub needed
125  else
126  a_ = 0;
127  }
