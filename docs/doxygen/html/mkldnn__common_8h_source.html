79  // Used as the key for MemoryReorder primitive reuse pool.

88  key.append(std::to_string(src_desc.format));
89  key.append(1, '_');

92  key.append(std::to_string(dst_desc.format));
93  key.append(1, '_');
94  key.append(std::to_string(dst_desc.data_type));
95  AddDimsToKey(key, dst_dims);

140 // Pool which allows for reuse of MKLDNN memory reorder primitives which are expensive to instantiate.
141 // To address thread safety, the primitives are stored in a map on thread local storage.
142 template <typename T>
143 class MemoryReorderPrimitivePool : public PrimitivePool<T> {
144  public:
145  static MemoryReorderPrimitivePool& GetInstance() {
146  static MemoryReorderPrimitivePool pool;
147  return pool;
148  }
149 
150  static MemoryReorderPrimitive* Get(const MemoryReorderParams& params) {
151  MemoryReorderPrimitive* primitive = static_cast<MemoryReorderPrimitive*>(
152  MemoryReorderPrimitivePool<T>::GetInstance().GetPrimitive(params.ToString()));
153  if (primitive == nullptr) {
154  auto reorder_primitive = std::make_unique<MemoryReorderPrimitive>(params);
155  primitive = reorder_primitive.get();
156  MemoryReorderPrimitivePool<T>::GetInstance().SetPrimitive(params.ToString(), std::move(reorder_primitive));
157  }
158  primitive->SetMemory(params);
159  return primitive;
160  }
161 
162  private:
163  MemoryReorderPrimitivePool() = default;
164  ~MemoryReorderPrimitivePool() = default;
165 };
