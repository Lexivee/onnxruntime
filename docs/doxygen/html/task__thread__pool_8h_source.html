151  void MainLoop(std::size_t index) {
152  while (running_) {
153  // Wait on condition variable while the task is empty and
154  // the pool is still running.
155  std::unique_lock<std::mutex> lock(mutex_);
156  while (tasks_.empty() && running_) {
157  condition_.wait(lock);
158  }
159 
160  // If pool is no longer running, break out of loop.
161  if (!running_) break;
162 
163  // Copy task locally and remove from the queue. This is
164  // done within its own scope so that the task object is
165  // destructed immediately after running the task. This is
166  // useful in the event that the function contains
167  // shared_ptr arguments bound via bind.
168  {
169  auto task = std::move(tasks_.front());
170  tasks_.pop();
171  // Decrement count, indicating thread is no longer available.
172  --available_;
173 
174  lock.unlock();
175 
176  // Run the task.
177  try {
178  if (task.run_with_id) {
179  task.with_id(index);
180  } else {
181  task.no_id();
182  }
183  } catch (const std::exception& /*ex*/) {
184  // LOGS_DEFAULT(ERROR) << "Exception running TaskThreadPool task: " << ex.what();
185  throw;
186  }
187 
188  // Update status of empty, maybe
189  // Need to recover the lock first
190  lock.lock();
191 
192  // Increment count, indicating thread is available.
193  ++available_;
194  if (tasks_.empty() && available_ == total_) {
195  complete_ = true;
196  completed_.notify_one();
197  }
198  }
199  } // while running_
