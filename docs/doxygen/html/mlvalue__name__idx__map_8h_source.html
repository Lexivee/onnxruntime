20  // Add MLValue name to map and return index associated with it.
21  // If entry already existed the existing index value is returned.
22  int Add(const std::string& name) {
23  auto it = map_.find(name);
24  if (it == map_.end()) {
25  int idx;

29  }

42  return common::Status::OK();
43  }
