11  GemmHelper(const TensorShape& left, bool trans_left, const TensorShape& right, bool trans_right, const TensorShape& bias) {
12  //dimension check
13  ONNXRUNTIME_ENFORCE(left.NumDimensions() == 2);
14  ONNXRUNTIME_ENFORCE(right.NumDimensions() == 2);
15 
16  if (trans_left) {
17  M_ = left[1];
18  K_ = left[0];
19  } else {
20  M_ = left[0];
21  K_ = left[1];
22  }
23 
24  int k_dim;
25  if (trans_right) {
26  N_ = right[0];
27  k_dim = 1;
28  } else {
29  N_ = right[1];
30  k_dim = 0;
31  }
32 
33  if (right[k_dim] != K_)
34  status_ = ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, INVALID_ARGUMENT,
35  "GEMM: Dimension mismatch, W: ",
36  right.ToString(),
37  " K: " + std::to_string(K_),
38  " N:" + std::to_string(N_));
39 
40  if (!IsValidBroadcast(bias, M_, N_))
41  status_ = common::Status(common::ONNXRUNTIME, common::INVALID_ARGUMENT, "Gemm: Invalid bias shape for broadcast");
42 
43  ONNXRUNTIME_ENFORCE(M_ > 0 && N_ > 0 && K_ > 0);
44  }

52  bool IsValidBroadcast(const TensorShape& shape, int64_t M, int64_t N) {
53  if (shape.NumDimensions() != 1 && shape.NumDimensions() != 2)
54  return false;
55  // shape is (1,) or (1, 1), or (,)
56  if (shape.Size() == 1)
57  return true;
58  // shape is (N,) or (1, N) or (M, 1)
59  // or (M, N), in last case no broadcast needed, but don't fail it
60  return ((shape.NumDimensions() == 1 && shape[0] == N) ||
61  (shape.NumDimensions() == 2 && shape[0] == M && (shape[1] == 1 || shape[1] == N)) ||
62  (shape.NumDimensions() == 2 && shape[0] == 1 && shape[1] == N));
63  }
