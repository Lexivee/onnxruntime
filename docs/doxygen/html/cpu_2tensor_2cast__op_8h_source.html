50 template <typename SrcType,
51  typename DstType>
52 inline void CastFloat16Data(const Tensor* in, Tensor* out, const TensorShape& shape, const AllocatorPtr& allocator) {
53  ONNXRUNTIME_ENFORCE(allocator != nullptr);
54  const int64_t len = shape.Size();
55  ONNXRUNTIME_ENFORCE(len > 0);
56  void* buffer = allocator->AllocArray(sizeof(float), len);
57  ONNXRUNTIME_ENFORCE(buffer);
58  Tensor tmp_tensor(DataTypeImpl::GetType<float>(), shape, buffer, allocator->Info(), nullptr);
59  if (std::is_same<SrcType, MLFloat16>::value) {
60  CastData<MLFloat16, float>(in, &tmp_tensor, shape); // first cast to float
61  CastData<float, DstType>(&tmp_tensor, out, shape); // then cast to the destination type.
62  } else if (std::is_same<DstType, MLFloat16>::value) {
63  CastData<SrcType, float>(in, &tmp_tensor, shape);
64  CastData<float, MLFloat16>(&tmp_tensor, out, shape);
65  }
66  allocator->Free(buffer);
67 }
