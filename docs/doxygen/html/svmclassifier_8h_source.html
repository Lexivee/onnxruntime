32  float kernel_dot(const T* A, int64_t a, const std::vector<float>& B, int64_t b, int64_t len, KERNEL k) const {
33  float sum = 0.f;
34  if (k == KERNEL::POLY) {
35  for (int64_t i = 0; i < len; i++) {
36  sum += B[b + i] * static_cast<float>(A[a + i]);
37  }
38  sum = gamma_ * sum + coef0_;
39  sum = std::pow(sum, degree_);
40  } else if (k == KERNEL::SIGMOID) {
41  for (int64_t i = 0; i < len; i++) {
42  sum += B[b + i] * static_cast<float>(A[a + i]);
43  }
44  sum = gamma_ * sum + coef0_;
45  sum = std::tanh(sum);
46  } else if (k == KERNEL::RBF) {
47  for (int64_t i = 0; i < len; i++) {
48  float val = static_cast<float>(A[a + i]) - B[b + i];
49  sum += (val * val);
50  }
51  sum = std::exp(-gamma_ * sum);
52  } else if (k == KERNEL::LINEAR) {
53  for (int64_t i = 0; i < len; i++) {
54  sum += B[b + i] * static_cast<float>(A[a + i]);
55  }
56  }
57  return sum;
58  }
59 
60  private:
61  KERNEL kernel_type_;
62  float gamma_;
63  float coef0_;
64  float degree_;
65 };
