27 class Notification {
28  public:
29  Notification() : notified_(false) {}
30  ~Notification() {
31  // In case the notification is being used to synchronize its own deletion,
32  // force any prior notifier to leave its critical section before the object
33  // is destroyed.
34  std::unique_lock<std::mutex> l(mu_);

37  void Notify() {
38  std::unique_lock<std::mutex> l(mu_);
39  assert(!HasBeenNotified());
40  notified_.store(true, std::memory_order_release);
41  cv_.notify_all();
42  }
43 
44  bool HasBeenNotified() const {
45  return notified_.load(std::memory_order_acquire);
46  }

48  void WaitForNotification() {
49  if (!HasBeenNotified()) {
50  std::unique_lock<std::mutex> l(mu_);
51  while (!HasBeenNotified()) {
52  cv_.wait(l);
53  }
54  }
55  }
56 
57  private:
58  friend bool WaitForNotificationWithTimeout(Notification* n,
59  int64_t timeout_in_us);
60  bool WaitForNotificationWithTimeout(int64_t timeout_in_us) {
61  bool notified = HasBeenNotified();
62  if (!notified) {
63  std::unique_lock<std::mutex> l(mu_);
64  do {
65  notified = HasBeenNotified();
66  } while (!notified &&
67  cv_.wait_for(l, std::chrono::microseconds(timeout_in_us)) !=
68  std::cv_status::timeout);
69  }
70  return notified;
71  }

74  std::condition_variable cv_; // signaled when notified_ becomes non-zero
75  std::atomic<bool> notified_; // mutations under mu_
76 };
