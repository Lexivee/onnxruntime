162 static inline float ml_inv_erf(float x) {
163  float sgn = x < 0 ? -1.0f : 1.0f;
164  x = (1 - x) * (1 + x);
165  float log = std::log(x);
166  float v = 2 / (3.14159f * 0.147f) + 0.5f * log;
167  float v2 = 1 / (0.147f) * log;
168  float v3 = -v + std::sqrt(v * v - v2);
169  x = sgn * std::sqrt(v3);
170  return x;
171 }

174 static inline void multiclass_probability(int64_t classcount, const std::vector<float>& r, std::vector<float>& p) {
175  int64_t sized2 = classcount * classcount;
176  std::vector<float> Q;
177  std::vector<float> Qp;
178  for (int64_t k = 0; k < sized2; k++) {
179  Q.push_back(0);
180  }
181  for (int64_t k = 0; k < classcount; k++) {
182  Qp.push_back(0);
183  }
184  float eps = 0.005f / static_cast<float>(classcount);
185  for (int64_t i = 0; i < classcount; i++) {
186  p[i] = 1.0f / static_cast<float>(classcount); // Valid if k = 1
187  for (int64_t j = 0; j < i; j++) {
188  Q[i * classcount + i] += r[j * classcount + i] * r[j * classcount + i];
189  Q[i * classcount + j] = Q[j * classcount + i];
190  }
191  for (int64_t j = i + 1; j < classcount; j++) {
192  Q[i * classcount + i] += r[j * classcount + i] * r[j * classcount + i];
193  Q[i * classcount + j] = -r[j * classcount + i] * r[i * classcount + j];
194  }
195  }
196  for (int64_t loop = 0; loop < 100; loop++) {
197  // stopping condition, recalculate QP,pQP for numerical accuracy
198  float pQp = 0;
199  for (int64_t i = 0; i < classcount; i++) {
200  Qp[i] = 0;
201  for (int64_t j = 0; j < classcount; j++) {
202  Qp[i] += Q[i * classcount + j] * p[j];
203  }
204  pQp += p[i] * Qp[i];
205  }
206  float max_error = 0;
207  for (int64_t i = 0; i < classcount; i++) {
208  float error = std::fabs(Qp[i] - pQp);
209  if (error > max_error) {
210  max_error = error;
211  }
212  }
213  if (max_error < eps) break;
214 
215  for (int64_t i = 0; i < classcount; i++) {
216  float diff = (-Qp[i] + pQp) / Q[i * classcount + i];
217  p[i] += diff;
218  pQp = (pQp + diff * (diff * Q[i * classcount + i] + 2 * Qp[i])) / (1 + diff) / (1 + diff);
219  for (int64_t j = 0; j < classcount; j++) {
220  Qp[j] = (Qp[j] + diff * Q[i * classcount + j]) / (1 + diff);
221  p[j] /= (1 + diff);
222  }
223  }
224  }
225 }

239 static inline void compute_softmax(std::vector<float>& values) {
240  std::vector<float> newscores;
241  // compute exp with negative number to be numerically stable
242  float v_max = -std::numeric_limits<float>::max();
243  for (float value : values) {
244  if (value > v_max)
245  v_max = value;
246  }
247  float this_sum = 0.f;
248  for (float value : values) {
249  float val2 = std::exp(value - v_max);
250  this_sum += val2;
251  newscores.push_back(val2);
252  }
253  for (int64_t k = 0; k < static_cast<int64_t>(values.size()); k++) {
254  values[k] = newscores[k] / this_sum;
255  }
256 }

258 //this function skips zero values (since exp(0) is non zero)
259 static inline void compute_softmax_zero(std::vector<float>& values) {
260  std::vector<float> newscores;
261  // compute exp with negative number to be numerically stable
262  float v_max = -std::numeric_limits<float>::max();
263  for (float value : values) {
264  if (value > v_max)
265  v_max = value;
266  }
267  float exp_neg_v_max = std::exp(-v_max);
268  float this_sum = 0.f;
269  for (float value : values) {
270  if (value > 0.0000001f || value < -0.0000001f) {
271  float val2 = std::exp(value - v_max);
272  this_sum += val2;
273  newscores.push_back(val2);
274  } else {
275  newscores.push_back(value * exp_neg_v_max);
276  }
277  }
278  for (int64_t k = 0; k < static_cast<int64_t>(values.size()); k++) {
279  values[k] = newscores[k] / this_sum;
280  }
281 }

283 static inline void write_scores(std::vector<float>& scores, POST_EVAL_TRANSFORM post_transform, int64_t write_index, Tensor* Z, int add_second_class) {
284  if (post_transform == POST_EVAL_TRANSFORM::PROBIT && scores.size() == 1) {
285  scores[0] = ml_sqrt2 * ml_inv_erf(2 * scores[0] - 1);
286  Z->template MutableData<float>()[write_index] = scores[0];
287  } else if (scores.size() >= 2) { //multiclass
288  if (post_transform == POST_EVAL_TRANSFORM::LOGISTIC) {
289  for (float& score : scores) {
290  score = ml_logit(score);
291  }
292  } else if (post_transform == POST_EVAL_TRANSFORM::SOFTMAX) {
293  compute_softmax(scores);
294  } else if (post_transform == POST_EVAL_TRANSFORM::SOFTMAX_ZERO) {
295  compute_softmax_zero(scores);
296  }
297  } else { //binary case
298  if (add_second_class == 0 && scores.size() == 1) { //0=all positive weights, winning class is positive
299  scores.push_back(scores[0]);
300  scores[0] = 1.f - scores[0]; //put opposite score in positive slot
301  } else if (add_second_class == 1 && scores.size() == 1) { //1 = all positive weights, winning class is negative
302  scores.push_back(scores[0]);
303  scores[0] = 1.f - scores[0]; //put opposite score in positive slot
304  } else if (add_second_class == 2 && scores.size() == 1) { //2 = mixed weights, winning class is positive
305  if (post_transform == POST_EVAL_TRANSFORM::LOGISTIC) {
306  scores.push_back(ml_logit(scores[0])); //ml_logit(scores[k]);
307  scores[0] = ml_logit(-scores[0]);
308  } else {
309  scores.push_back(scores[0]);
310  scores[0] = -scores[0];
311  }
312  } else if (add_second_class == 3 && scores.size() == 1) { //3 = mixed weights, winning class is negative
313  if (post_transform == POST_EVAL_TRANSFORM::LOGISTIC) {
314  scores.push_back(ml_logit(scores[0])); //ml_logit(scores[k]);
315  scores[0] = ml_logit(-scores[0]);
316  } else {
317  scores.push_back(-scores[0]);
318  }
319  }
320  }
321  for (float score : scores) {
322  Z->template MutableData<float>()[write_index] = score;
323  write_index++;
324  }
325 }
