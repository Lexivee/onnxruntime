10 // MemPatternPlanner is used to trace allocation/free steps
11 // in a single iteration, record the pattern and cached for
12 // future request if they have the same input shape.
13 class MemPatternPlanner {
14  public:
15  MemPatternPlanner() = default;

17  void TraceAllocation(int ml_value_idx, size_t size) {
18  if (size == 0) {
19  allocs_.emplace_back(ml_value_idx, MemoryBlock(0, 0));
20  return;
21  }
22 
23  size_t current = 0;
24  size_t waste_bytes = std::numeric_limits<size_t>::max();
25  size_t best_offset = 0;
26  if (!blocks_.empty()) {
27  auto last_block = allocs_[*blocks_.rbegin()];
28  best_offset = last_block.block_.offset_ + last_block.block_.size_;
29  }
30 
31  std::list<int>::iterator best_fit_it = blocks_.end();
32  for (auto it = blocks_.begin(); it != blocks_.end(); it++) {
33  if (allocs_[*it].block_.offset_ >= current) {
34  auto gap = allocs_[*it].block_.offset_ - current;
35  if (gap >= size && (gap - size) < waste_bytes) {
36  best_fit_it = it;
37  waste_bytes = gap - size;
38  best_offset = current;
39  }
40  }
41  current = allocs_[*it].block_.offset_ + allocs_[*it].block_.size_;
42  }
43 
44  allocs_.emplace_back(ml_value_idx, MemoryBlock(best_offset, size));
45  buffer_size = std::max(buffer_size, best_offset + size);
46  blocks_.insert(best_fit_it, (static_cast<int>(allocs_.size()) - 1));
47  }
48 
49  void TraceFree(int ml_value_index) {
50  for (auto it = blocks_.begin(); it != blocks_.end(); it++) {
51  if (allocs_[*it].index_ == ml_value_index) {
52  blocks_.erase(it);
53  break;
54  }
55  }
56  }

58  MemoryPattern GenerateMemPattern() {
59  MemoryPattern pattern;
60  pattern.peak_size_ = buffer_size;
61  for (auto& alloc : allocs_) {
62  pattern.patterns_[alloc.index_] = alloc.block_;
63  }
64 
65  return pattern;
66  }
