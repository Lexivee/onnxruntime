90  return nullptr;

92  MLValue* p_ml_value = nullptr;
93  ONNXRUNTIME_ENFORCE(GetOrCreateOutputMLValue(index, p_ml_value).IsOK());
94  return p_ml_value ? p_ml_value->GetMutable<T>() : nullptr;
95  }
