55 // The base class for rewrite rule. A rewrite rule represents a semantics-preserving
56 // transformation of a computation-graph. It can be used to represent, for example,
57 // the elimination of operators that serve as no-ops (for example, dropout during
58 // inference), as well as inlining of "function" definitions or the dual (replacing
59 // a complex expression by an equivalent function-call). Unlike the more general
60 // IGraphTransformer, a rewrite-rule is applied at a single node, representing the
61 // root of an expression that is rewritten.
62 class RewriteRule {
63  public:
64  RewriteRule(const std::string& name, const std::string& desc)
65  : name_(name), desc_(desc) {
66  }
67

69

71  const std::string& Name() const noexcept {
72  return name_;
73  }
74 
75  // An description of this rewrite rule.
76  const std::string& Description() const noexcept {
77  return desc_;
78  }

80  // If the condition of the rule is satisfied, apply the rule.
81  ::onnxruntime::common::Status CheckConditionAndApply(GraphEditor* graph_editor, Node* node, bool* modified) {
82  return SatisfyCondition(*node) ? Apply(graph_editor, node, modified) : Status::OK();
83  }
84 
85  private:
86  ONNXRUNTIME_DISALLOW_COPY_ASSIGNMENT_AND_MOVE(RewriteRule);
87 
88  const std::string name_;
89  const std::string desc_;
90 
91  // The rewrite rule is applied if the condition function returns true. This can include
92  // a more complex pattern matching (conditions on the ascending or descending nodes of the
93  // node for which this rule was triggered) or some other properties of the nodes.
94  virtual bool SatisfyCondition(const Node& node) = 0;
95 
96  // Apply the rewrite rule to a specific node.

98  // from the input-value due to rewriting.
99  // The return value of "modified" indicates if the graph was modified or not.
100  virtual ::onnxruntime::common::Status Apply(GraphEditor* graph_editor, Node* node, bool* modified) = 0;
101 };
