21 class ExecutionProviders {

25  common::Status Add(const std::string& provider_id, std::unique_ptr<IExecutionProvider> p_exec_provider) {
26  // make sure there are no issues before we change any internal data structures
27  if (provider_idx_map_.find(provider_id) != provider_idx_map_.end()) {
28  auto status = ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, FAIL, "Provider ", provider_id, " has already been registered.");
29  LOGS_DEFAULT(ERROR) << status.ErrorMessage();
30  return status;
31  }
32 
33  for (const auto& allocator : p_exec_provider->GetAllocatorMap()) {
34  if (allocator_idx_map_.find(allocator->Info()) != allocator_idx_map_.end()) {
35  auto status = ONNXRUNTIME_MAKE_STATUS(ONNXRUNTIME, FAIL, allocator->Info(), " allocator already registered.");
36  LOGS_DEFAULT(ERROR) << status.ErrorMessage();
37  return status;
38  }
39  }
40 
41  // index that provider will have after insertion
42  auto new_provider_idx = exec_providers_.size();
43 
44  ONNXRUNTIME_IGNORE_RETURN_VALUE(provider_idx_map_.insert({provider_id, new_provider_idx}));
45 
46  for (const auto& allocator : p_exec_provider->GetAllocatorMap()) {
47  ONNXRUNTIME_IGNORE_RETURN_VALUE(allocator_idx_map_.insert({allocator->Info(), new_provider_idx}));
48  }
49 
50  exec_providers_.push_back(std::move(p_exec_provider));
51 
52  return Status::OK();
53  }
54

56  auto node = graph.GetNode(idx);
57  ONNXRUNTIME_ENFORCE(node != nullptr, "Node with id ", idx, " was not found in graph.");
