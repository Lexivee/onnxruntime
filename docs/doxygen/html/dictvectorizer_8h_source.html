12 template <typename AttrType, typename TargetType>
13 class DictVectorizerOp final : public OpKernel {
14  public:
15  DictVectorizerOp(const OpKernelInfo& info) : OpKernel(info) {
16  //In some stupid models, the vocabulary could have duplicated elements.
17  //We must support that, otherwise some tests will be break.
18  ONNXRUNTIME_ENFORCE(info.GetAttrs(std::is_same<AttrType, std::string>::value ? "string_vocabulary" : "int64_vocabulary", vocabulary_).IsOK());
19  }
20  common::Status Compute(OpKernelContext* ctx) const override {
21  auto map = ctx->Input<std::map<AttrType, TargetType> >(0);
22  std::vector<int64_t> dims{1, static_cast<int64_t>(vocabulary_.size())};
23  auto Y = ctx->Output(0, TensorShape(dims));
24  auto* y_data = Y->template MutableData<TargetType>();
25  for (size_t i = 0, end = vocabulary_.size(); i < end; ++i) {
26  auto index = map->find(vocabulary_[i]);
27  if (index != map->end()) {
28  *y_data++ = index->second;
29  } else {
30  //Any keys not present in the input dictionary, will be zero in the output array
31  *y_data++ = TargetType();
32  }
33  }
34  return Status::OK();
35  }
36 
37  std::vector<AttrType> vocabulary_;
38 };
