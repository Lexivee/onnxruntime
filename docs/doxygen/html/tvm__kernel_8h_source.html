11 // TVMScheduleCreator is the function that create a tvm schedule based on given TVM graph.

16 typedef tvm::runtime::Module (*TVMModuleBuilder)(tvm::Schedule schedule, tvm::BuildConfig config, tvm::Array<tvm::Tensor> args, std::vector<std::string>& target_func_names);

18 template <TVMScheduleCreator S, TVMModuleBuilder M>
19 class TVMKernel : public OpKernel {
20  public:
21  explicit TVMKernel(const OpKernelInfo& info) : OpKernel(info), tvm_values_(nullptr), dl_tensors_(nullptr), tvm_type_codes_(nullptr) {
22  auto& node = info.node();
23  ONNXRUNTIME_ENFORCE(node.NodeType() == Node::Type::Fused);
24  auto func = node.GetFunctionBody();
25  const onnxruntime::Graph& func_body = func->Body();
26  //1. compile the onnxruntime Graph to tvm graph. This step is common for all hardware, and provided by onnxruntime framework.
27  tvm_graph_ = CompileToTVM(func_body, node.GetExecutionProviderType());
28  //2. create schedule for tvm graph, this step is depends on the execution provider/hardware.
29  auto s = S(tvm_graph_);
30  //3. Build module
31  std::vector<tvm::Tensor> tvm_args;
32  for (auto& t : tvm_graph_.inputs_) {
33  tvm_args.push_back(t.tvm_tensor_);
34  }
35  for (auto& t : tvm_graph_.outputs_) {
36  tvm_args.push_back(t.tvm_tensor_);
37  }
38 
39  std::vector<std::string> func_names;
40  tvm_module_ = M(s, tvm::build_config(), tvm_args, func_names);
41  //TODO: do we have case that need more than 1 evaluation function?
42  evaluate_func_ = tvm_module_.GetFunction(func_names[0]);
43  //4. prepare args according to the type
44  n_args_ = tvm_args.size();
45  tvm_values_ = new TVMValue[n_args_];
46  tvm_type_codes_ = new int[n_args_];
47  dl_tensors_ = new DLTensor[n_args_];
48  int i = 0;
49  for (auto& tensor : tvm_graph_.inputs_) {
50  tvm_type_codes_[i] = kNDArrayContainer;
51  dl_tensors_[i].ctx = tensor.ctx_;
52  dl_tensors_[i].dtype = tensor.dtype_;
53  dl_tensors_[i].strides = nullptr;
54  dl_tensors_[i].byte_offset = 0;
55  tvm_values_[i].v_handle = &dl_tensors_[i];
56  i++;
57  }
58 
59  for (auto& tensor : tvm_graph_.outputs_) {
60  tvm_type_codes_[i] = kNDArrayContainer;
61  dl_tensors_[i].ctx = tensor.ctx_;
62  dl_tensors_[i].dtype = tensor.dtype_;
63  dl_tensors_[i].strides = nullptr;
64  dl_tensors_[i].byte_offset = 0;
65  tvm_values_[i].v_handle = &dl_tensors_[i];
66  i++;
67  }
68  ONNXRUNTIME_ENFORCE(i == n_args_);
69  }

80  virtual Status Compute(OpKernelContext* context) const override {
81  for (int i = 0; i < tvm_graph_.inputs_.size(); ++i) {
82  auto t = context->Input<Tensor>(i);
83  dl_tensors_[i].data = const_cast<Tensor*>(t)->MutableDataRaw();
84  dl_tensors_[i].ndim = static_cast<int>(t->Shape().NumDimensions());
85  dl_tensors_[i].shape = dl_tensors_[i].ndim > 0 ? const_cast<int64_t*>(&(t->Shape().GetDims()[0])) : nullptr;
86  }
87 
88  int num_inputs = static_cast<int>(tvm_graph_.inputs_.size());
89 
90  for (int i = 0; i < tvm_graph_.outputs_.size(); ++i) {
91  //TODO: we need to have a shape inference function that could calculate output shape based on the symbolic formular in tvm
92  //We could build that function as part of tvm module, or reuse the shape inference in onnx funciton.
93  //Here for testing purpose, assume the output shape is same to input shape.
94  auto t = context->Output(i, GetOutputShape(context, i));
95  dl_tensors_[num_inputs + i].data = t->MutableDataRaw();
96  dl_tensors_[num_inputs + i].ndim = static_cast<int>(t->Shape().NumDimensions());
97  dl_tensors_[num_inputs + i].shape = dl_tensors_[i].ndim > 0 ? const_cast<int64_t*>(&(t->Shape().GetDims()[0])) : nullptr;
98  }
99 
100  tvm::TVMArgs tvm_args(&tvm_values_[0], &tvm_type_codes_[0], static_cast<int>(n_args_));
101  tvm::TVMRetValue rvalue;
102  try {
103  evaluate_func_.CallPacked(tvm_args, &rvalue);
104  } catch (std::exception ex) {
105  return Status(onnxruntime::common::ONNXRUNTIME, onnxruntime::common::FAIL, "TVM run failed.");
106  }
107  if (rvalue.type_code() != kNull) {
108  return Status(onnxruntime::common::ONNXRUNTIME, onnxruntime::common::FAIL, "TVM return not null"); // TODO: get error code.
109  } else {
110  return Status::OK();
111  }
112  }
