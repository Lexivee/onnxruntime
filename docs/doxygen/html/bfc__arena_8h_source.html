125  // A ChunkHandle is an index into the chunks_ vector in BFCAllocator
126  // kInvalidChunkHandle means an invalid chunk
127  using ChunkHandle = size_t;
128  static const size_t kInvalidChunkHandle = static_cast<size_t>(-1);

130  using BinNum = int;

134  // Chunks point to memory. Their prev/next pointers form a
135  // doubly-linked list of addresses sorted by base address that
136  // must be contiguous. Chunks contain information about whether
137  // they are in use or whether they are free, and contain a pointer
138  // to the bin they are in.
139  struct Chunk {
140  size_t size = 0; // Full size of buffer.
141 
142  // We sometimes give chunks that are larger than needed to reduce
143  // fragmentation. requested_size keeps track of what the client
144  // actually wanted so we can understand whether our splitting
145  // strategy is efficient.
146  size_t requested_size = 0;
147 
148  // allocation_id is set to -1 when the chunk is not in use. It is assigned a
149  // value greater than zero before the chunk is returned from
150  // AllocateRaw, and this value is unique among values assigned by
151  // the parent allocator.
152  int64_t allocation_id = -1;
153  void* ptr = nullptr; // pointer to granted subbuffer.
154 
155  // If not kInvalidChunkHandle, the memory referred to by 'prev' is directly
156  // preceding the memory used by this chunk. E.g., It should start
157  // at 'ptr - prev->size'
158  ChunkHandle prev = kInvalidChunkHandle;
159 
160  // If not kInvalidChunkHandle, the memory referred to by 'next' is directly
161  // following the memory used by this chunk. E.g., It should be at
162  // 'ptr + size'
163  ChunkHandle next = kInvalidChunkHandle;
164 
165  // What bin are we in?
166  BinNum bin_num = kInvalidBinNum;
167 
168  bool in_use() const { return allocation_id != -1; }
169 
170  std::string DebugString(BFCArena* a, bool recurse) {
171  std::ostringstream ss;
172  ss << " Size: " << size << " | Requested Size: " << requested_size << " | in_use: " << in_use();
173  if (recurse && prev != BFCArena::kInvalidChunkHandle) {
174  Chunk* p = a->ChunkFromHandle(prev);
175  ss << ", prev: " << p->DebugString(a, false);
176  }
177 
178  if (recurse && next != BFCArena::kInvalidChunkHandle) {
179  Chunk* n = a->ChunkFromHandle(next);
180  ss << ", next: " << n->DebugString(a, false);
181  }
182  return ss.str();
183  }
184  };

191  struct ChunkComparator {
192  explicit ChunkComparator(BFCArena* allocator)
193  : allocator_(allocator) {}
194 
195  // Sort first by size and then use pointer address as a tie breaker.
196  bool operator()(const ChunkHandle ha,
197  const ChunkHandle hb) const {
198  const Chunk* a = allocator_->ChunkFromHandle(ha);
199  const Chunk* b = allocator_->ChunkFromHandle(hb);
200  if (a->size != b->size) {
201  return a->size < b->size;
202  }
203  return a->ptr < b->ptr;
204  }
205 
206  private:
207  BFCArena* allocator_; // The parent allocator
208  };

221  // AllocationRegion maps pointers to ChunkHandles for a single
222  // contiguous memory region.
223  //
224  // This class is thread-compatible.
225  class AllocationRegion {
226  public:
227  AllocationRegion(void* ptr, size_t memory_size)
228  : ptr_(ptr),
229  memory_size_(memory_size),
230  end_ptr_(
231  static_cast<void*>(static_cast<char*>(ptr_) + memory_size_)) {
232  ONNXRUNTIME_ENFORCE(0 == memory_size % kMinAllocationSize);
233  const size_t n_handles =
234  (memory_size + kMinAllocationSize - 1) / kMinAllocationSize;
235  handles_ = new ChunkHandle[n_handles];
236  for (size_t i = 0; i < n_handles; i++) {
237  handles_[i] = kInvalidChunkHandle;
238  }
239  }
240 
241  AllocationRegion() = default;
242 
243  ~AllocationRegion() { delete[] handles_; }
244 
245  AllocationRegion(AllocationRegion&& other) { Swap(other); }
246 
247  AllocationRegion& operator=(AllocationRegion&& other) {
248  Swap(other);
249  return *this;
250  }
251 
252  void* ptr() const { return ptr_; }
253  void* end_ptr() const { return end_ptr_; }
254  size_t memory_size() const { return memory_size_; }
255  ChunkHandle get_handle(const void* p) const {
256  return handles_[IndexFor(p)];
257  }
258  void set_handle(const void* p, ChunkHandle h) { handles_[IndexFor(p)] = h; }
259  void erase(const void* p) { set_handle(p, kInvalidChunkHandle); }
260 
261  private:
262  void Swap(AllocationRegion& other) {
263  std::swap(ptr_, other.ptr_);
264  std::swap(memory_size_, other.memory_size_);
265  std::swap(end_ptr_, other.end_ptr_);
266  std::swap(handles_, other.handles_);

269  int IndexFor(const void* p) const {
270  std::uintptr_t p_int = reinterpret_cast<std::uintptr_t>(p);
271  std::uintptr_t base_int = reinterpret_cast<std::uintptr_t>(ptr_);
272  ONNXRUNTIME_ENFORCE(p_int >= base_int);
273  ONNXRUNTIME_ENFORCE(p_int < base_int + memory_size_);
274  return static_cast<int>(((p_int - base_int) >> kMinAllocationBits));
275  }
276 
277  // Metadata about the allocation region.
278  void* ptr_ = nullptr;
279  size_t memory_size_ = 0;
280  void* end_ptr_ = nullptr;
281 
282  // Array of size "memory_size / kMinAllocationSize". It is
283  // indexed by (p-base) / kMinAllocationSize, contains ChunkHandle
284  // for the memory allocation represented by "p"
285  ChunkHandle* handles_ = nullptr;
286 
287  ONNXRUNTIME_DISALLOW_ASSIGNMENT(AllocationRegion);

404  int Log2FloorNonZeroSlow(uint64_t n) {
405  int r = 0;
406  while (n > 0) {
407  r++;
408  n >>= 1;
409  }
410  return r - 1;
411  }
412 
413  // Returns floor(log2(n)).
414  int Log2FloorNonZero(uint64_t n) {
415 #if defined(__GNUC__)
416  return 63 ^ __builtin_clzll(n);
417 #elif defined(PLATFORM_WINDOWS)
418  unsigned long index;
419 #if defined(_WIN64)
420  _BitScanReverse64(&index, n);
421 #else
422  auto high = static_cast<unsigned long>(n >> 32);
423  if (_BitScanReverse(&index, high) > 0) {
424  index += 32;
425  } else {
426  auto low = static_cast<unsigned long>((n << 32) >> 32);
427  _BitScanReverse(&index, low);
428  }
429 #endif
430  return index;
431 #else
432  return Log2FloorNonZeroSlow(n);
433 #endif
434  }
